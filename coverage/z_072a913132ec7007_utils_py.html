21def build_install_command( 

22 extras: str, 

23 *, 

24 git_ref: str | None = None, 

25 version: str | None = None, 

26 pre_release: bool = False, 

27 python: Path | None = None, 

28 upgrade: bool = False, 

29 verbose: bool = False, 

30) -> tuple[str, dict[str, str]]: 

31 """Build a pip/uv install shell command for reachy-mini. 

32 

33 For a *git_ref* install the command chains three steps: 

34 

35 1. ``--force-reinstall --no-deps --no-cache-dir`` - reinstall the package 

36 itself without the dependencies. 

37 2. ``check`` - check if dependencies need to be updated. 

38 3. ``--upgrade --upgrade-strategy only-if-needed`` - if dependencies need to be updated, upgrade them. 

39 

40 Args: 

41 extras: Pip extras string, e.g. ``"gstreamer"`` or 

42 ``"wireless-version, gstreamer"``. 

43 git_ref: If set, install from this GitHub tag/branch. 

44 version: If set (and *git_ref* is ``None``), pin to this PyPI version. 

45 pre_release: If ``True`` (and neither *git_ref* nor *version* is set), 

46 add ``--pre`` to allow pre-release versions. 

47 python: Target Python interpreter for an external venv. 

48 Uses ``uv`` when available, otherwise the venv's own ``pip``. 

49 If ``None``, uses the current environment's ``pip``. 

50 upgrade: If ``True``, add the ``--upgrade`` flag. 

51 verbose: If ``True``, add ``-vvv`` flag (only on step 2 for git ref). 

52 

53 Returns: 

54 A tuple of the form ``(command, extra_env)`` where *command* is a shell string and *extra_env* contains any additional environment variables needed for the install. 

55 

56 """ 

57 # --- Base command (pip or uv) --- 

58 if python is not None: 

59 if _check_uv_available(): 

60 base = ["uv", "pip", "install", "--python", str(python)] 

61 logger.info(f"Using uv with python: {python}") 

62 else: 

63 base = [str(python.parent / "pip"), "install"] 

64 logger.info(f"Using pip from venv: {python.parent / 'pip'}") 

65 else: 

66 if _check_uv_available(): 

67 base = ["uv", "pip", "install"] 

68 logger.info("Using uv pip with current environment") 

69 else: 

70 base = ["pip", "install"] 

71 logger.info("Using pip with current environment") 

72 

73 if verbose: 

74 base.append("-vvv") 

75 

76 # --- Package, extra args & env --- 

77 if git_ref: 

78 logger.info(f"Installing from git ref: {git_ref}") 

79 git_url = f"git+https://github.com/{GITHUB_REPO}.git@{git_ref}" 

80 git_package = f"reachy-mini[{extras}] @ {git_url}" 

81 # Step 1: force reinstall the package without the dependencies 

82 step1 = shlex.join(base + [git_package, "--force-reinstall", "--no-deps", "--no-cache-dir"]) 

83 # Step 2: check if dependencies need to be updated 

84 check_base = [arg if arg != "install" else "check" for arg in base] 

85 step2 = shlex.join(check_base) 

86 # Step 3: update dependencies if needed 

87 step3_args = base + [f"reachy-mini[{extras}]", "--upgrade"] 

88 if not _check_uv_available(): 

89 step3_args += ["--upgrade-strategy", "only-if-needed"] 

90 step3 = shlex.join(step3_args) 

91 cmd = f"{step1} && {step2} || {step3}" 

92 logger.info(f"Git ref install: {cmd}") 

93 extra_env: dict[str, str] = {"GIT_LFS_SKIP_SMUDGE": "1"} 

94 return cmd, extra_env 

95 

96 logger.info(f"Installing from PyPI: {version if version else 'latest pre-release' if pre_release else 'latest stable'}") 

97 package = f"reachy-mini[{extras}]" 

98 if version: 

99 package = f"{package}=={version}" 

100 extra_args = [] 

101 if pre_release: 

102 extra_args.append("--pre") 

103 if upgrade: 

104 extra_args.append("--upgrade") 

105 extra_env = {} 

106 

107 cmd = shlex.join(base + [package] + extra_args) 

108 logger.info(f"Install command: {cmd}") 

109 return cmd, extra_env

112async def call_logger_wrapper( 

113 command: str, 

114 logger: logging.Logger, 

115 env: dict[str, str] | None = None, 

116) -> None: 

117 """Run a shell command asynchronously, streaming stdout and stderr to logger in real time. 

118 

119 Args: 

120 command: Shell command string. 

121 logger: logger object with .info and .error methods 

122 env: Optional environment variables dict. If None, inherits current environment. 

123 

124 """ 

125 logger.info(f"Running: {command}") 

126 process = await asyncio.create_subprocess_shell( 

127 command, 

128 stdout=asyncio.subprocess.PIPE, 

129 stderr=asyncio.subprocess.PIPE, 

130 env={**os.environ, **env} if env else None, 

131 cwd=Path.home(), 

132 )

134 async def stream_output( 

135 stream: asyncio.StreamReader, 

136 log_func: Callable[[str], None], 

137 ) -> None: 

138 while True: 

139 line = await stream.readline() 

140 if not line: 

141 break 

142 log_func(line.decode().rstrip()) 

143 

144 tasks = [] 

145 if process.stdout is not None: 

146 tasks.append(asyncio.create_task(stream_output(process.stdout, logger.info))) 

147 if process.stderr is not None: 

148 tasks.append(asyncio.create_task(stream_output(process.stderr, logger.error))) 

149 

150 await asyncio.gather(*tasks) 

151 await process.wait()
