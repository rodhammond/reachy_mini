92 """Relay signaling messages between central server (HTTP/SSE) and local GStreamer (WebSocket). 

93 

94 This class maintains connections to both a central signaling server (for remote access) 

95 and the local GStreamer WebRTC signaling server, relaying messages between them. 

96 

97 The relay is designed to be robust: 

98 - Automatically reconnects on connection failures 

99 - Refreshes the HF token on each reconnection attempt 

100 - Can be notified of token changes for immediate reconnection 

101 """

355 async def stop(self) -> None: 

356 """Stop the relay service.""" 

357 logger.info("[Central Relay] Stopping relay service...") 

358 self._running = False 

359 

360 # Wake up any waiting in the thread's event loop 

361 if self._thread_loop and self._thread_loop.is_running(): 

362 self._thread_loop.call_soon_threadsafe(self._token_updated.set) 

363 

364 # Wait for thread to finish 

365 if self._thread and self._thread.is_alive(): 

366 self._thread.join(timeout=5.0) 

367 if self._thread.is_alive(): 

368 logger.warning("[Central Relay] Thread did not stop within timeout")

371

373 if self._robot_app_lock is not None: 

374 self._robot_app_lock.set_remote_eviction_handler(None) 

375 self._robot_app_lock.release_remote() 

376 

377 self._set_state(RelayState.STOPPED, "Relay stopped")

454 reason, 

455 )

462 async def _notify_peer_session_failed( 

463 self, 

464 peer_id: str, 

465 reason: str, 

466 diagnostic: dict[str, Any], 

467 ) -> None: 

468 """Inner coroutine for ``notify_peer_session_failed``. 

469 

470 Runs on the relay's own thread loop. Looks up the mapping, 

471 emits ``endSession`` to central, and clears the bookkeeping 

472 for this session. Local GStreamer is intentionally NOT 

473 notified here: ``webrtcbin`` already considers the peer dead 

474 (that's what triggered the watchdog) and emitting 

475 ``endSession`` to local would race ``consumer-removed``, 

476 producing a confusing pair of "session ended" log lines for 

477 the same UUID. 

478 """ 

479 # Walk the mapping in one pass and capture the central id 

480 # before mutating the dicts, so we can log a single coherent 

481 # line even if a concurrent ``endSession`` from the other 

482 # direction is racing us. 

483 local_session_id = peer_id 

484 central_session_id = self._local_to_central_session.get(local_session_id) 

485 if central_session_id is None: 

486 logger.warning( 

487 "[Central Relay] notify_peer_session_failed: no central session " 

488 "mapped for peer_id=%s (reason=%s, diagnostic=%s); session may " 

489 "have already been torn down", 

490 peer_id, 

491 reason, 

492 diagnostic, 

493 ) 

494 return 

495 

496 logger.error( 

497 "[Central Relay] daemon-side WebRTC failure: peer_id=%s -> " 

498 "central_session=%s reason=%s diagnostic=%s", 

499 peer_id, 

500 central_session_id, 

501 reason, 

502 diagnostic, 

503 ) 

504 

505 try: 

506 await self._send_to_central( 

507 { 

508 "type": "endSession", 

509 "sessionId": central_session_id, 

510 "reason": reason, 

511 } 

512 ) 

513 except Exception: 

514 logger.warning( 

515 "[Central Relay] Failed to notify central of peer session failure", 

516 exc_info=True, 

517 ) 

518 

519 # Clean up our half of the bookkeeping. ``_consumer_removed`` 

520 # on the GStreamer side will run shortly after and clean up 

521 # the local-side mappings too, but we drop our entries 

522 # eagerly to avoid an inconsistent window where central 

523 # already considers the session done but the relay still has 

524 # stale state. 

525 self._local_to_central_session.pop(local_session_id, None) 

526 self._central_to_local_session.pop(central_session_id, None) 

527 self._session_to_local_peer.pop(central_session_id, None) 

528 if central_session_id in self._pending_central_sessions: 

529 self._pending_central_sessions.remove(central_session_id) 

530 if ( 

531 self._robot_app_lock is not None 

532 and not self._central_to_local_session 

533 and not self._pending_central_sessions 

534 ): 

535 self._robot_app_lock.release_remote()

585 async def _close_connections(self) -> None: 

586 """Close all connections. 

587 

588 Reentrant-safe: three converging paths can call this on the 

589 same event loop in quick succession (see ``self._closing`` in 

590 ``__init__`` for the full triad). The guard short-circuits the 

591 second and third calls so a single teardown can't double-close 

592 an already-closing aiohttp session or interleave halfway 

593 through the dict clears. 

594 """ 

595 if self._closing: 

596 return 

597 self._closing = True 

598 try: 

599 if self._http_session: 

600 try: 

601 await self._http_session.close() 

602 except Exception: 

603 pass 

604 self._http_session = None 

605 

606 if self._local_ws: 

607 try: 

608 await self._local_ws.close() 

609 except Exception: 

610 pass 

611 self._local_ws = None 

612 

613 # Clear session state 

614 self._central_peer_id = None 

615 self._local_peer_id = None 

616 self._session_to_local_peer.clear() 

617 self._pending_central_sessions.clear()

629 def _reconnect_delay(self) -> float: 

630 """Return the next reconnect wait using capped exponential backoff. 

631 

632 The delay grows as ``RECONNECT_INTERVAL * 2 ** (attempts - 1)`` per 

633 consecutive failed attempt, capped at ``MAX_RECONNECT_INTERVAL``, with 

634 a small additive jitter to de-synchronize reconnects across a fleet. 

635 ``_connection_attempts`` resets to 0 on a successful connection, so the 

636 backoff restarts from ``RECONNECT_INTERVAL`` after recovery. The 

637 ``_token_updated`` event still short-circuits the wait, so a login or 

638 token refresh reconnects immediately regardless of the backoff. 

639 """ 

640 exponent = min(max(self._connection_attempts - 1, 0), 20) 

641 base: float = min(RECONNECT_INTERVAL * (2**exponent), MAX_RECONNECT_INTERVAL) 

642 jitter: float = random.uniform(0.0, base * RECONNECT_BACKOFF_JITTER) 

643 return base + jitter

645 async def _run_loop(self) -> None: 

646 """Maintain connections and relay messages.""" 

647 logger.info("[Central Relay] _run_loop started") 

648 

649 try: 

650 # Small yield to allow event loop to process other events 

651 await asyncio.sleep(0) 

652 logger.info("[Central Relay] Starting connection attempts") 

653 

654 while self._running: 

655 had_exception = False 

656 try: 

657 await self._connect_and_relay() 

658 except asyncio.CancelledError: 

659 # CancelledError at this layer has two possible sources: 

660 # 

661 # 1. `stop()` flipped `self._running` to False. This is the 

662 # shutdown path and we must re-raise so the thread exits. 

663 # 2. An in-flight `_close_connections()` (triggered by 

664 # `force_reconnect` / `update_token`) cancelled a task 

665 # that aiohttp was holding the cancellation for - e.g. 

666 # a `session.get(...)` wedged inside `_resolve_host` on 

667 # a flaky network. aiohttp propagates that cancellation 

668 # up through `_handle_central_sse`, which lands here. 

669 # In that case we very much want to stay in the loop and 

670 # reconnect - killing the thread here means every 

671 # subsequent `/refresh-relay` POST is a no-op because 

672 # there's no loop left to service the token_updated 

673 # event. 

674 # 

675 # `self._running` is our authoritative signal for (1). If 

676 # it's still True, treat the cancellation as a reconnect 

677 # request and loop around. 

678 if not self._running: 

679 logger.info( 

680 "[Central Relay] _run_loop cancelled (stop requested)" 

681 ) 

682 raise 

683 logger.info( 

684 "[Central Relay] Connect attempt cancelled mid-flight " 

685 "(likely from force_reconnect / token update); restarting loop" 

686 ) 

687 had_exception = True 

688 self._set_state( 

689 RelayState.RECONNECTING, 

690 "Restarting after cancelled connect", 

691 ) 

692 except Exception as e: 

693 logger.warning( 

694 f"[Central Relay] Connection attempt failed with exception: {type(e).__name__}: {e}" 

695 ) 

696 had_exception = True 

697 self._connection_attempts += 1 

698 if self._connection_attempts <= 3: 

699 self._set_state( 

700 RelayState.RECONNECTING, f"Connection failed: {e}" 

701 ) 

702 else: 

703 self._set_state( 

704 RelayState.ERROR, 

705 f"Connection failed after {self._connection_attempts} attempts: {e}", 

706 ) 

707 

708 if self._running and not had_exception and self._state == RelayState.ERROR: 

709 # Clean return but ERROR (e.g. 401) - back off like a failure. 

710 had_exception = True 

711 self._connection_attempts += 1 

712 

713 if self._running and had_exception: 

714 # Only wait after connection failures, not after normal returns 

715 # (e.g., when token update triggered reconnection) 

716 self._token_updated.clear() 

717 try: 

718 await asyncio.wait_for( 

719 self._token_updated.wait(), 

720 timeout=self._reconnect_delay(), 

721 ) 

722 except asyncio.TimeoutError: 

723 pass 

724 

725 except asyncio.CancelledError: 

726 logger.info("[Central Relay] Run loop cancelled (stop requested)") 

727 raise 

728 except Exception as e: 

729 logger.error(f"[Central Relay] Unexpected error in run loop: {e}") 

730 raise

944 @staticmethod 

945 def _negotiate_heartbeat_interval(welcome_msg: dict[str, Any]) -> float: 

946 """Pick the cadence at which we re-emit setPeerStatus. 

947 

948 Priority order: 

949 1. ``recommended_heartbeat_interval_seconds`` field from the 

950 welcome (canonical signal: central advertises whatever 

951 cadence its server-side ``LEASE_SECONDS`` env tunes to, 

952 typically ``LEASE / 3``). 

953 2. ``lease_seconds`` field from the welcome divided by 3 

954 (older centrals that expose lease but not the recommended 

955 cadence directly). 

956 3. ``HEARTBEAT_DEFAULT_INTERVAL`` for pre-negotiation 

957 centrals that expose neither. 

958 

959 Rungs 1 and 2 are passed through ``_clamp_heartbeat_interval`` 

960 so a misconfigured central can neither ask us to spam (say, 

961 0.1 s) nor lull us into a cadence so slow we'd be evicted 

962 before the next heartbeat fires (say, 600 s). 

963 """ 

964 raw = welcome_msg.get("recommended_heartbeat_interval_seconds") 

965 if isinstance(raw, (int, float)) and raw > 0: 

966 return _clamp_heartbeat_interval(float(raw)) 

967 lease = welcome_msg.get("lease_seconds") 

968 if isinstance(lease, (int, float)) and lease > 0: 

969 return _clamp_heartbeat_interval(float(lease) / 3.0) 

970 return HEARTBEAT_DEFAULT_INTERVAL

972 async def _heartbeat_loop(self) -> None: 

973 """Re-emit setPeerStatus periodically to keep the central lease alive. 

974 

975 Central runs a TTL sweeper: a producer that does not generate 

976 inbound traffic (POST /send) for more than ``LEASE_SECONDS`` is 

977 evicted from its in-memory tables, even if the SSE channel is 

978 perfectly healthy. Half-open sockets (Wi-Fi yanked, NAT 

979 rebinding, captive portal sleep) are the prime offenders - 

980 the local TCP stack absorbs server-pushed keepalives silently 

981 for minutes and the daemon never realises it's already a ghost. 

982 

983 This loop is the daemon-side half of the contract: 

984 * Cadence is set by ``_negotiate_heartbeat_interval`` from 

985 the SSE welcome and stored on ``_heartbeat_interval_seconds``. 

986 We re-read it on every iteration so the negotiated value 

987 propagates as soon as the welcome lands, even though this 

988 loop is spawned in parallel with the SSE handler that 

989 processes welcome (the first iteration uses the default, 

990 every subsequent one uses the negotiated value). 

991 * The ``await sleep(interval)`` happens BEFORE each re-emit, 

992 so the first heartbeat fires one interval after the loop 

993 starts (the post-welcome registration just refreshed 

994 ``last_seen``; firing again immediately would be wasteful). 

995 * Each re-emit reuses ``_producer_status_payload`` so we 

996 stay byte-identical to the registration payload. Central 

997 is idempotent under repeated setPeerStatus from the same peer. 

998 * A failed heartbeat is logged at DEBUG, not ERROR: the 

999 split-brain detector (``_producer_health_loop``) is the 

1000 authoritative recovery path. If central truly dropped us, 

1001 the next robot-status poll will notice and trigger a 

1002 ``force_reconnect()``; redundant teardown from this loop 

1003 would only fight that mechanism. 

1004 * Exits on cancellation when ``_connect_and_relay``'s 

1005 FIRST_COMPLETED race tears all sibling tasks down. 

1006 """ 

1007 last_logged_interval: Optional[float] = None 

1008 

1009 while self._running: 

1010 interval = self._heartbeat_interval_seconds 

1011 if interval != last_logged_interval: 

1012 logger.info("[Central Relay] heartbeat loop interval=%.1fs", interval) 

1013 last_logged_interval = interval 

1014 

1015 try: 

1016 await asyncio.sleep(interval) 

1017 except asyncio.CancelledError: 

1018 raise 

1019 

1020 if self._state != RelayState.CONNECTED: 

1021 # Reconnecting / errored: nothing to refresh. Once we 

1022 # land back in CONNECTED, this loop has already been 

1023 # cancelled and respawned with a freshly negotiated 

1024 # interval. 

1025 continue 

1026 

1027 try: 

1028 await self._send_to_central(self._producer_status_payload()) 

1029 except asyncio.CancelledError: 

1030 raise 

1031 except Exception as e: 

1032 logger.debug("[Central Relay] heartbeat re-emit failed: %s", e) 

1033

1044 HTTP session, or transient HTTP/network failure). Callers 

1045 treat ``None`` as "don't update health counters" so a brief 

1046 central hiccup never triggers a needless reconnect storm. 

1047 

1048 """

1050 return None

1052 headers = {"Authorization": f"Bearer {self.hf_token}"}

1054 try: 

1055 async with self._http_session.get( 

1056 url, headers=headers, timeout=timeout 

1057 ) as response: 

1058 if response.status != 200: 

1059 logger.debug( 

1060 "[Central Relay] producer health check returned HTTP %s", 

1061 response.status,

1065 except asyncio.CancelledError:

1070 except Exception as e: 

1071 logger.warning( 

1072 "[Central Relay] producer health check unexpected error: %s", e 

1073 ) 

1074 return None 

1075 

1076 robots = data.get("robots") if isinstance(data, dict) else None 

1077 if not isinstance(robots, list): 

1078 return None

1082 async def _producer_health_loop(self) -> None: 

1083 """Self-heal split-brain states by polling /api/robot-status. 

1084 

1085 Without this, a relay whose ``setPeerStatus`` was cancelled 

1086 mid-flight (token rotation, DNS hiccup, transient HTTP error) 

1087 keeps an SSE channel open and reports state=CONNECTED even 

1088 though central has no record of it as a producer. From the 

1089 outside the robot looks online but no client can call it - the 

1090 only recovery used to be SSH+systemctl or a manual POST to 

1091 /api/hf-auth/refresh-relay. 

1092 

1093 Loop semantics: 

1094 * Wait ``PRODUCER_HEALTH_CHECK_INITIAL_DELAY`` after start so 

1095 the post-welcome ``setPeerStatus`` round-trip has time to 

1096 land in central's index. 

1097 * Every ``PRODUCER_HEALTH_CHECK_INTERVAL`` seconds, ask 

1098 central whether we are still listed. 

1099 * Tolerate up to ``PRODUCER_HEALTH_CHECK_MAX_MISSES`` consecutive 

1100 "not listed" answers before self-triggering a 

1101 ``force_reconnect()``. This avoids reconnect storms on 

1102 harmless central blips. 

1103 * Exit on cancellation (``_connect_and_relay``'s 

1104 FIRST_COMPLETED race tears us down on every reconnect). 

1105 """ 

1106 try: 

1107 await asyncio.sleep(PRODUCER_HEALTH_CHECK_INITIAL_DELAY) 

1108 except asyncio.CancelledError: 

1109 raise 

1110 

1111 consecutive_misses = 0 

1112 while self._running: 

1113 try: 

1114 await asyncio.sleep(PRODUCER_HEALTH_CHECK_INTERVAL) 

1115 except asyncio.CancelledError: 

1116 raise 

1117 

1118 if self._state != RelayState.CONNECTED: 

1119 # Don't probe while we're already reconnecting / in error - 

1120 # there's nothing useful we can do about the listing state 

1121 # until the SSE handshake settles again. 

1122 consecutive_misses = 0 

1123 continue 

1124 

1125 listed = await self._is_listed_as_producer() 

1126 if listed is True: 

1127 consecutive_misses = 0 

1128 continue 

1129 if listed is None: 

1130 # Transient error - keep our streak intact rather than 

1131 # padding it with noise. 

1132 continue 

1133 

1134 consecutive_misses += 1 

1135 logger.warning( 

1136 "[Central Relay] producer not listed on central " 

1137 "(miss %s/%s, peer_id=%s)", 

1138 consecutive_misses, 

1139 PRODUCER_HEALTH_CHECK_MAX_MISSES, 

1140 self._central_peer_id, 

1141 ) 

1142 if consecutive_misses >= PRODUCER_HEALTH_CHECK_MAX_MISSES: 

1143 logger.warning( 

1144 "[Central Relay] split-brain detected, forcing reconnect" 

1145 ) 

1146 # `force_reconnect` schedules a teardown via the thread 

1147 # loop; this task gets cancelled by the FIRST_COMPLETED 

1148 # race in `_connect_and_relay` and the next iteration 

1149 # of `_run_loop` re-spawns us with a fresh peer_id. 

1150 await self.force_reconnect() 

1151 return

1153 async def _handle_central_sse(self) -> None: 

1154 """Handle SSE events from central server.""" 

1155 if not self._http_session: 

1156 return 

1157 

1158 # Token goes in the Authorization header, never in the URL — 

1159 # keeps it out of HF Space access logs and intermediate proxies. 

1160 events_url = f"{self.central_uri}/events" 

1161 headers = {"Authorization": f"Bearer {self.hf_token}"} 

1162 

1163 try: 

1164 # Use timeout for the initial connection 

1165 timeout = aiohttp.ClientTimeout( 

1166 total=None, connect=10, sock_read=SSE_READ_TIMEOUT 

1167 ) 

1168 async with self._http_session.get( 

1169 events_url, headers=headers, timeout=timeout 

1170 ) as response: 

1171 if response.status == 401: 

1172 self._set_state( 

1173 RelayState.ERROR, "Authentication failed - token may be invalid" 

1174 ) 

1175 return 

1176 elif response.status != 200: 

1177 self._set_state( 

1178 RelayState.ERROR, 

1179 f"Central server returned HTTP {response.status}", 

1180 ) 

1181 return 

1182 

1183 # Connection successful - will set CONNECTED after welcome message 

1184 self._connection_attempts = 0 

1185 

1186 # Read lines with timeout to detect dead connections 

1187 while self._running: 

1188 try: 

1189 line = await asyncio.wait_for( 

1190 response.content.readline(), timeout=SSE_READ_TIMEOUT 

1191 ) 

1192 except asyncio.TimeoutError: 

1193 logger.warning( 

1194 f"[Central Relay] SSE read timeout after {SSE_READ_TIMEOUT}s - connection may be dead" 

1195 ) 

1196 self._set_state( 

1197 RelayState.RECONNECTING, 

1198 "Connection timeout, reconnecting...", 

1199 ) 

1200 return 

1201 

1202 if not line: 

1203 # Empty line means connection closed 

1204 logger.info("[Central Relay] SSE connection closed by server") 

1205 return 

1206 

1207 line_str = line.decode("utf-8").strip() 

1208 

1209 if line_str.startswith("data:"): 

1210 data = line_str[5:].strip() 

1211 if data: 

1212 try: 

1213 msg = json.loads(data) 

1214 await self._process_central_message(msg) 

1215 except json.JSONDecodeError: 

1216 logger.warning( 

1217 f"[Central Relay] Invalid JSON from central: {data[:100]}" 

1218 ) 

1219 

1220 except asyncio.CancelledError: 

1221 raise 

1222 except aiohttp.ClientError as e: 

1223 self._set_state(RelayState.ERROR, f"Central server unreachable: {e}") 

1224 except Exception as e: 

1225 logger.error(f"[Central Relay] Error in central SSE: {e}") 

1226 finally:

1311 async def _process_central_message(self, msg: dict[str, Any]) -> None: 

1312 """Process a message from the central server.""" 

1313 msg_type = msg.get("type", "") 

1314 logger.debug(f"[Central Relay] Received from central server: type={msg_type}") 

1315 

1316 if msg_type == "welcome": 

1317 # Received our peer ID from central server 

1318 self._central_peer_id = msg.get("peerId") 

1319 self._heartbeat_interval_seconds = self._negotiate_heartbeat_interval(msg) 

1320 logger.info( 

1321 "[Central Relay] central welcome received peer_id=%s; " 

1322 "registering as producer name=%r, heartbeat=%.1fs", 

1323 self._central_peer_id, 

1324 self.robot_name, 

1325 self._heartbeat_interval_seconds, 

1326 ) 

1327 

1328 # Register as producer FIRST, then flip to CONNECTED. If we 

1329 # set CONNECTED before producer registration, observers (UI, 

1330 # mobile app, /relay-status pollers) can see "connected" while 

1331 # central does not yet know we are a producer for this user, 

1332 # which produces the desync described in /refresh-relay's 

1333 # docstring. 

1334 await self._send_to_central(self._producer_status_payload()) 

1335 

1336 self._set_state( 

1337 RelayState.CONNECTED, f"Remote access enabled as '{self.robot_name}'" 

1338 ) 

1339 

1340 elif msg_type == "list": 

1341 # Ignore list messages - we're a producer 

1342 pass 

1343 

1344 elif msg_type == "startSession": 

1345 # A client wants to connect - forward to local GStreamer 

1346 client_peer_id: str = msg.get("peerId", "") 

1347 session_id: Optional[str] = msg.get("sessionId") 

1348 logger.info( 

1349 f"[Central Relay] Received session request from remote client peer_id={client_peer_id} session_id={session_id}" 

1350 ) 

1351 

1352 # Safety net: the central server is supposed to gate concurrent sessions, 

1353 # but if one slips through (e.g. older central without the gate), enforce 

1354 # single-session-at-a-time here so we never run two clients against one robot. 

1355 if self._central_to_local_session or self._pending_central_sessions: 

1356 logger.warning( 

1357 f"[Central Relay] Rejecting session {session_id}: a session is already active/pending" 

1358 ) 

1359 if session_id: 

1360 await self._send_to_central( 

1361 { 

1362 "type": "endSession", 

1363 "sessionId": session_id, 

1364 "reason": "robot_busy_local", 

1365 } 

1366 ) 

1367 return 

1368 

1369 # Gate on the robot lock: if a local Python app is running, the 

1370 # lock will refuse our acquire. We also acquire proactively here 

1371 # so a concurrent local-app start can't sneak in between the 

1372 # check and the session handoff to local GStreamer. 

1373 if self._robot_app_lock is not None: 

1374 # holder_name is generic because central already tracks the 

1375 # real consumer app name (via setPeerStatus meta) for its 

1376 # own rejection messages; the daemon-side lock just needs 

1377 # to know that *something* remote holds it. 

1378 if not self._robot_app_lock.try_acquire_remote("remote"): 

1379 logger.warning( 

1380 f"[Central Relay] Rejecting session {session_id}: robot lock is held locally" 

1381 ) 

1382 if session_id: 

1383 await self._send_to_central( 

1384 { 

1385 "type": "endSession", 

1386 "sessionId": session_id, 

1387 "reason": "robot_busy_local_app", 

1388 } 

1389 ) 

1390 return 

1391 

1392 # Store session mapping 

1393 if session_id: 

1394 # Check if we already have this session (duplicate request) 

1395 if session_id in self._session_to_local_peer: 

1396 logger.warning( 

1397 f"[Central Relay] Duplicate session request for session_id={session_id}, ignoring" 

1398 ) 

1399 return 

1400 

1401 self._session_to_local_peer[session_id] = client_peer_id 

1402 self._pending_central_sessions.append(session_id) 

1403 logger.info( 

1404 f"[Central Relay] Pending sessions: {len(self._pending_central_sessions)}, tracked sessions: {len(self._session_to_local_peer)}" 

1405 ) 

1406 

1407 # Request list of local producers to start session 

1408 await self._send_to_local({"type": "list"}) 

1409 

1410 elif msg_type == "peer": 

1411 # SDP/ICE from client - relay to local GStreamer 

1412 central_session_id = msg.get("sessionId") 

1413 if central_session_id and self._local_ws: 

1414 # Translate central session ID to local session ID 

1415 local_session_id = self._central_to_local_session.get( 

1416 central_session_id 

1417 ) 

1418 if not local_session_id: 

1419 logger.warning( 

1420 f"[Central Relay] No local session mapping found for central_session_id={central_session_id}" 

1421 ) 

1422 return 

1423 

1424 local_msg = { 

1425 "type": "peer", 

1426 "sessionId": local_session_id, 

1427 } 

1428 if "sdp" in msg: 

1429 local_msg["sdp"] = msg["sdp"] 

1430 if "ice" in msg: 

1431 local_msg["ice"] = msg["ice"] 

1432 

1433 logger.debug( 

1434 f"[Central Relay] Relaying peer message: central_session={central_session_id} -> local_session={local_session_id}" 

1435 ) 

1436 await self._send_to_local(local_msg) 

1437 

1438 elif msg_type == "endSession": 

1439 central_session_id = msg.get("sessionId") 

1440 if central_session_id: 

1441 logger.info( 

1442 f"[Central Relay] Session ended from central: central_session_id={central_session_id}" 

1443 ) 

1444 self._session_to_local_peer.pop(central_session_id, None) 

1445 # Also remove from pending if it never got started 

1446 if central_session_id in self._pending_central_sessions: 

1447 self._pending_central_sessions.remove(central_session_id) 

1448 # Translate and forward to local 

1449 local_session_id = self._central_to_local_session.pop( 

1450 central_session_id, None 

1451 ) 

1452 if local_session_id: 

1453 self._local_to_central_session.pop(local_session_id, None) 

1454 logger.info( 

1455 f"[Central Relay] Forwarding endSession to local: local_session_id={local_session_id}" 

1456 ) 

1457 await self._send_to_local( 

1458 {"type": "endSession", "sessionId": local_session_id} 

1459 ) 

1460 logger.info( 

1461 f"[Central Relay] After cleanup - pending: {len(self._pending_central_sessions)}, active: {len(self._central_to_local_session)}" 

1462 ) 

1463 # If no sessions remain, release the robot lock. 

1464 if ( 

1465 self._robot_app_lock is not None 

1466 and not self._central_to_local_session 

1467 and not self._pending_central_sessions 

1468 ): 

1469 self._robot_app_lock.release_remote() 

1470 

1471 elif msg_type == "peerStatusChanged": 

1472 # Another peer changed status - ignore for producers 

1473 pass

1475 async def _process_local_message(self, msg: dict[str, Any]) -> None: 

1476 """Process a message from local GStreamer signaling.""" 

1477 msg_type = msg.get("type", "") 

1478 logger.debug(f"[Central Relay] Received from local GStreamer: type={msg_type}") 

1479 

1480 if msg_type == "welcome": 

1481 # Received our peer ID from local GStreamer 

1482 self._local_peer_id = msg.get("peerId") 

1483 logger.info( 

1484 f"[Central Relay] Connected to local GStreamer signaling server peer_id={self._local_peer_id}" 

1485 )

1487 # Signal that local connection is verified 

1488 if hasattr(self, "_local_welcome_received"): 

1489 self._local_welcome_received.set()

1492 await self._send_to_local(

1494 "type": "setPeerStatus", 

1495 "roles": ["listener"], 

1496 "meta": {"name": "central-relay"},

1500 elif msg_type == "list": 

1501 # List of local producers 

1502 producers = msg.get("producers", []) 

1503 if producers: 

1504 self._local_producer_id = producers[0].get("id") 

1505 logger.debug( 

1506 f"[Central Relay] Local GStreamer producer found: producer_id={self._local_producer_id}"

1509 # Only start sessions for PENDING requests, not all tracked sessions 

1510 for central_session_id in list(self._pending_central_sessions): 

1511 logger.info( 

1512 f"[Central Relay] Starting local session for pending central_session={central_session_id}" 

1513 ) 

1514 await self._send_to_local( 

1515 { 

1516 "type": "startSession", 

1517 "peerId": self._local_producer_id, 

1518 } 

1519 ) 

1520 

1521 elif msg_type == "peerStatusChanged": 

1522 peer_id = msg.get("peerId") 

1523 roles = msg.get("roles", []) 

1524 if "producer" in roles: 

1525 self._local_producer_id = peer_id 

1526 logger.debug( 

1527 f"[Central Relay] Local GStreamer producer registered: producer_id={peer_id}" 

1528 ) 

1529 

1530 elif msg_type == "sessionStarted": 

1531 local_session_id: Optional[str] = msg.get("sessionId") 

1532 logger.info( 

1533 f"[Central Relay] Local GStreamer session started: local_session_id={local_session_id}" 

1534 )

1536 # Map local session ID to the pending central session ID 

1537 if self._pending_central_sessions and local_session_id: 

1538 central_session_id = self._pending_central_sessions.pop(0) 

1539 self._local_to_central_session[local_session_id] = central_session_id 

1540 self._central_to_local_session[central_session_id] = local_session_id 

1541 logger.info( 

1542 f"[Central Relay] Session mapping established: local_session={local_session_id} <-> central_session={central_session_id}" 

1543 ) 

1544 

1545 elif msg_type == "peer": 

1546 # SDP/ICE from local GStreamer - relay to central 

1547 local_session_id_peer: Optional[str] = msg.get("sessionId") 

1548 if local_session_id_peer: 

1549 # Translate local session ID to central session ID 

1550 central_session_id_peer: Optional[str] = ( 

1551 self._local_to_central_session.get(local_session_id_peer) 

1552 ) 

1553 if not central_session_id_peer: 

1554 logger.warning( 

1555 f"[Central Relay] No central session mapping found for local_session_id={local_session_id_peer}" 

1556 ) 

1557 return 

1558 

1559 # Build message with translated session ID 

1560 central_msg: dict[str, Any] = { 

1561 "type": "peer", 

1562 "sessionId": central_session_id_peer, 

1563 } 

1564 if "sdp" in msg: 

1565 central_msg["sdp"] = msg["sdp"] 

1566 if "ice" in msg: 

1567 central_msg["ice"] = msg["ice"] 

1568 

1569 logger.debug( 

1570 f"[Central Relay] Relaying peer message: local_session={local_session_id_peer} -> central_session={central_session_id_peer}" 

1571 ) 

1572 await self._send_to_central(central_msg) 

1573 

1574 elif msg_type == "endSession": 

1575 local_session_id_end: Optional[str] = msg.get("sessionId") 

1576 if local_session_id_end: 

1577 logger.info( 

1578 f"[Central Relay] Session ended from local: local_session_id={local_session_id_end}" 

1579 ) 

1580 # Translate and forward to central 

1581 central_session_id_end: Optional[str] = ( 

1582 self._local_to_central_session.pop(local_session_id_end, None) 

1583 ) 

1584 if central_session_id_end: 

1585 self._central_to_local_session.pop(central_session_id_end, None) 

1586 self._session_to_local_peer.pop(central_session_id_end, None) 

1587 logger.info( 

1588 f"[Central Relay] Forwarding endSession to central: central_session_id={central_session_id_end}" 

1589 ) 

1590 await self._send_to_central( 

1591 {"type": "endSession", "sessionId": central_session_id_end} 

1592 ) 

1593 logger.info( 

1594 f"[Central Relay] After cleanup - pending: {len(self._pending_central_sessions)}, active: {len(self._central_to_local_session)}"

1598 self._robot_app_lock is not None 

1599 and not self._central_to_local_session 

1600 and not self._pending_central_sessions

1724async def notify_force_reconnect() -> bool: 

1725 """Force the central signaling relay to drop and reconnect right now. 

1726 

1727 Drops the relay's current connection and re-registers with the

1730 ``/central-robot-status`` returns ``robots: []`` — central no 

1731 longer lists this robot as a producer for the authenticated user 

1732 while the relay still holds its SSE channel open. From the outside 

1733 this manifests as "the robot is online but no one can call it" 

1734 until someone restarts the daemon. 

1735 

1736 Common triggers: 

1737 - the relay attached with a token that has since been rotated; 

1738 - a transient error during ``setPeerStatus`` went unnoticed. 

1739

1741 ``old_token == new_token`` early-return and always reconnects, 

1742 which is exactly what callers of ``POST /api/hf-auth/refresh-relay`` 

1743 need (their whole reason for invoking is the token did not change 

1744 but the producer registration was lost). Goes through the relay's 

1745 own reconnect path — works with any token shape currently stored 

1746 (raw user tokens, OAuth access tokens) without re-validation. 

1747 

1748 Returns: 

1749 ``True`` if a reconnect was actually kicked off, ``False`` if 

1750 there was no relay instance to reconnect (e.g. Lite-only build, 

1751 daemon not yet up). HTTP callers (mobile auto-heal flow) must 

1752 surface this distinction so they can stop waiting on a 

1753 reconnect that will never happen. 

1754 

1755 """ 

1756 if _relay_instance is None: 

1757 logger.debug("[Central Relay] No relay instance, ignoring force reconnect") 

1758 return False
