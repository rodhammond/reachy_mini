462 def update_target_head_joints_from_ik( 

463 self, 

464 pose: Annotated[NDArray[np.float64], (4, 4)] | None = None, 

465 body_yaw: float | None = None, 

466 ) -> None: 

467 """Update the target head joint positions from inverse kinematics. 

468 

469 Args: 

470 pose (np.ndarray): 4x4 pose matrix representing the head pose. 

471 body_yaw (float): The yaw angle of the body, used to adjust the head pose. 

472 

473 """ 

474 if pose is None: 

475 pose = ( 

476 self.target_head_pose 

477 if self.target_head_pose is not None 

478 else np.eye(4) 

479 ) 

480 

481 if body_yaw is None: 

482 body_yaw = self.target_body_yaw if self.target_body_yaw is not None else 0.0 

483 

484 aim = self._tracking_aim 

485 if aim is not None and self._tracking_weight > 0.0: 

486 weight = min(max(self._tracking_weight, 0.0), 1.0) 

487 pose = linear_pose_interpolation(pose, aim, weight) 

488 

489 # Compose speech wobbler offsets (if any) before IK 

490 if any(o != 0.0 for o in self._speech_offsets): 

491 x_m, y_m, z_m, roll_r, pitch_r, yaw_r = self._speech_offsets 

492 offset_pose = create_head_pose( 

493 x=x_m, 

494 y=y_m, 

495 z=z_m, 

496 roll=roll_r, 

497 pitch=pitch_r, 

498 yaw=yaw_r, 

499 degrees=False, 

500 ) 

501 pose = compose_world_offset(pose, offset_pose) 

502 

503 # Compute the inverse kinematics to get the head joint positions 

504 joints = self.head_kinematics.ik(pose, body_yaw=body_yaw) 

505 if joints is None or np.any(np.isnan(joints)): 

506 raise ValueError("WARNING: Collision detected or head pose not achievable!") 

507 

508 # update the target head pose and body yaw 

509 self._last_target_head_pose = pose 

510 self._last_target_body_yaw = body_yaw 

511 

512 self.target_head_joint_positions = joints

748 async def play_move( 

749 self, 

750 move: Move, 

751 play_frequency: float = 100.0, 

752 initial_goto_duration: float = 0.0, 

753 audio_lead_s: float = 0.0, 

754 cancel_token: Optional[_PlaybackCancelToken] = None, 

755 ) -> None: 

756 """Asynchronously play a Move. 

757 

758 Args: 

759 move (Move): The Move object to be played. 

760 play_frequency (float): The frequency at which to evaluate the move (in Hz). 

761 initial_goto_duration (float): Duration for an initial goto to the move's starting position. If 0.0, no initial goto is performed. 

762 audio_lead_s (float): How many seconds the audio (if any) starts BEFORE the motion. Positive values compensate for the constant GStreamer playbin latency on the robot so the audio reaches the speaker at the same moment the actuator starts moving. Negative values delay audio relative to motion. No-op when the move has no sound_path. Default 0. 

763 cancel_token (_PlaybackCancelToken, optional): If provided, the inner loop polls ``cancel_token.cancelled`` every tick and exits when flipped. Used by ``_async_play_uploaded_move`` to wire ``cancel_move`` to a specific upload_id; direct callers (goto_target, etc.) pass None and stay non-cancellable. 

764 

765 """ 

766 if not self._try_start_move(): 

767 self.logger.warning("Ignoring play_move request: another move is running.") 

768 return 

769 

770 try: 

771 if initial_goto_duration > 0.0: 

772 start_head_pose, start_antennas_positions, start_body_yaw = ( 

773 move.evaluate(0.0) 

774 ) 

775 await self.goto_target( 

776 head=start_head_pose, 

777 antennas=start_antennas_positions, 

778 duration=initial_goto_duration, 

779 body_yaw=start_body_yaw, 

780 ) 

781 sleep_period = 1.0 / play_frequency 

782 

783 # Sound handoff. audio_lead_s shifts the audio start 

784 # relative to the motion loop: 

785 # > 0: audio starts first, motion follows after the wait 

786 # < 0: motion starts first, audio follows 

787 # = 0: kick off audio just before entering the loop (legacy behaviour) 

788 if move.sound_path is not None and audio_lead_s > 0: 

789 self.play_sound(str(move.sound_path)) 

790 await asyncio.sleep(audio_lead_s) 

791 elif move.sound_path is not None and audio_lead_s == 0: 

792 self.play_sound(str(move.sound_path)) 

793 

794 t0 = time.time() 

795 # Negative audio_lead_s: schedule sound to fire mid-loop, 

796 # |audio_lead_s| seconds after t0. We hold a reference to 

797 # the background task so the finally block can cancel it 

798 # if the motion loop exits before the sleep elapses (short 

799 # move + big negative lead would otherwise leak a task that 

800 # plays audio after the move has ended). 

801 delayed_sound_task: Optional["asyncio.Task[None]"] = None 

802 if move.sound_path is not None and audio_lead_s < 0: 

803 sound_path_str = str(move.sound_path) 

804 

805 async def _delayed_sound() -> None: 

806 try: 

807 await asyncio.sleep(-audio_lead_s) 

808 except asyncio.CancelledError: 

809 return 

810 if cancel_token is not None and cancel_token.cancelled: 

811 return 

812 self.play_sound(sound_path_str) 

813 

814 delayed_sound_task = asyncio.create_task(_delayed_sound()) 

815 try: 

816 while time.time() - t0 < move.duration: 

817 if cancel_token is not None and cancel_token.cancelled: 

818 self.logger.info("play_move cancelled, exiting playback loop") 

819 break 

820 t = time.time() - t0 

821 

822 head, antennas, body_yaw = move.evaluate(t) 

823 if head is not None: 

824 self.set_target_head_pose(head) 

825 if body_yaw is not None: 

826 self.set_target_body_yaw(body_yaw) 

827 if antennas is not None: 

828 self.set_target_antenna_joint_positions(antennas) 

829 

830 elapsed = time.time() - t0 - t 

831 if elapsed < sleep_period: 

832 await asyncio.sleep(sleep_period - elapsed) 

833 else: 

834 await asyncio.sleep(0.001) 

835 finally: 

836 # Don't leak the delayed-sound task past the loop. Two 

837 # cases this matters: the move duration was shorter 

838 # than |audio_lead_s|, or the loop was cancelled before 

839 # the sleep elapsed. 

840 if delayed_sound_task is not None and not delayed_sound_task.done(): 

841 delayed_sound_task.cancel() 

842 finally: 

843 self._end_move()

880 async def goto_joint_positions( 

881 self, 

882 head_joint_positions: list[float] 

883 | None = None, # [yaw, stewart_platform x 6] length 7 

884 antennas_joint_positions: list[float] 

885 | None = None, # [right_angle, left_angle] length 2 

886 duration: float = 0.5, # Duration in seconds for the movement 

887 method: InterpolationTechnique = InterpolationTechnique.MIN_JERK, # can be "linear", "minjerk", "ease_in_out" or "cartoon", default is "minjerk" 

888 ) -> None: 

889 """Asynchronously go to a target head joint positions and/or antennas joint positions using joint space interpolation, in "duration" seconds. 

890 

891 Go to a target head joint positions and/or antennas joint positions using joint space interpolation, in "duration" seconds. 

892 

893 Args: 

894 head_joint_positions (Optional[List[float]]): List of head joint positions in radians (length 7). 

895 antennas_joint_positions (Optional[List[float]]): List of antennas joint positions in radians (length 2). 

896 duration (float): Duration of the movement in seconds. Default is 0.5 seconds. 

897 method (str): Interpolation method to use ("linear", "minjerk", "ease_in_out", "cartoon"). Default is "minjerk". 

898 

899 Raises: 

900 ValueError: If neither head_joint_positions nor antennas_joint_positions are provided, or if duration is not positive. 

901 

902 """ 

903 if duration <= 0.0: 

904 raise ValueError( 

905 "Duration must be positive and non-zero. Use set_target() for immediate position setting." 

906 ) 

907 

908 start_head = np.array(self.get_present_head_joint_positions()) 

909 start_antennas = np.array(self.get_present_antenna_joint_positions()) 

910 

911 target_head = ( 

912 np.array(head_joint_positions) 

913 if head_joint_positions is not None 

914 else start_head 

915 ) 

916 target_antennas = ( 

917 np.array(antennas_joint_positions) 

918 if antennas_joint_positions is not None 

919 else start_antennas 

920 ) 

921 

922 t0 = time.time() 

923 while time.time() - t0 < duration: 

924 t = time.time() - t0 

925 

926 interp_time = time_trajectory(t / duration, method=method) 

927 

928 head_joint = start_head + (target_head - start_head) * interp_time 

929 antennas_joint = ( 

930 start_antennas + (target_antennas - start_antennas) * interp_time 

931 ) 

932 

933 self.set_target_head_joint_positions(head_joint) 

934 self.set_target_antenna_joint_positions(antennas_joint) 

935 await asyncio.sleep(0.01)

1180 _, _, dist_to_sleep_pose = distance_between_poses( 

1181 self.get_current_head_pose(), self.SLEEP_HEAD_POSE 

1182 ) 

1183 _, _, dist_to_init_pose = distance_between_poses( 

1184 self.get_current_head_pose(), self.INIT_HEAD_POSE 

1185 )

1679 def _evict_stale_uploads(self) -> None: 

1680 """Drop in-progress upload slots older than ``_upload_ttl_s``. 

1681 

1682 Called opportunistically on every new upload_move_start so we 

1683 never accumulate orphans (client crashed mid-upload, dropped 

1684 the data channel, etc.). 

1685 """ 

1686 now = time.time() 

1687 stale = [ 

1688 uid for uid, ts in self._upload_ts.items() if now - ts > self._upload_ttl_s 

1689 ] 

1690 for uid in stale: 

1691 self._upload_chunks.pop(uid, None) 

1692 self._upload_meta.pop(uid, None) 

1693 self._upload_ts.pop(uid, None) 

1694 self.logger.warning(f"upload_move: evicted stale slot {uid} (TTL exceeded)") 

1695 

1696 # All upload_* handlers are fire-and-forget. The client pipelines 

1697 # chunks at line rate (relying on SCTP's ordered, reliable delivery) 

1698 # and the daemon silently drops failed slots. The eventual 

1699 # play_uploaded_move broadcast will surface a "no such upload" 

1700 # error if anything went wrong during upload. 

1701 

1702 def _handle_upload_start(self, cmd: UploadMoveStartCmd) -> None: 

1703 self._evict_stale_uploads() 

1704 if len(self._upload_chunks) >= self._upload_max_active_slots: 

1705 self.logger.warning( 

1706 f"upload_move_start: refusing {cmd.upload_id}, too many active slots" 

1707 ) 

1708 return 

1709 # Sending start twice for the same id resets the slot, which 

1710 # lets a client retry after a transient send failure without 

1711 # needing to allocate a new id. 

1712 self._upload_chunks[cmd.upload_id] = [] 

1713 self._upload_meta[cmd.upload_id] = { 

1714 "total_chunks": cmd.total_chunks, 

1715 "description": cmd.description, 

1716 "estimated_duration_s": cmd.estimated_duration_s, 

1717 "encoding": cmd.encoding, 

1718 } 

1719 self._upload_ts[cmd.upload_id] = time.time() 

1720 

1721 def _handle_upload_chunk(self, cmd: UploadMoveChunkCmd) -> None: 

1722 slot = self._upload_chunks.get(cmd.upload_id) 

1723 meta = self._upload_meta.get(cmd.upload_id) 

1724 if slot is None or meta is None: 

1725 self.logger.warning( 

1726 f"upload_move_chunk: no slot {cmd.upload_id}, dropping chunk {cmd.chunk_index}" 

1727 ) 

1728 return 

1729 expected_index = len(slot) 

1730 if cmd.chunk_index != expected_index: 

1731 # Out-of-order on an ordered SCTP transport means a client 

1732 # bug. Drop the slot so the next start can recover cleanly. 

1733 self.logger.warning( 

1734 f"upload_move_chunk: out-of-order on {cmd.upload_id} " 

1735 f"(expected {expected_index}, got {cmd.chunk_index}); dropping slot" 

1736 ) 

1737 self._upload_chunks.pop(cmd.upload_id, None) 

1738 self._upload_meta.pop(cmd.upload_id, None) 

1739 self._upload_ts.pop(cmd.upload_id, None) 

1740 return 

1741 if cmd.chunk_index >= meta["total_chunks"]: 

1742 self.logger.warning( 

1743 f"upload_move_chunk: index {cmd.chunk_index} exceeds declared "

1969 def _handle_upload_finish(self, cmd: UploadMoveFinishCmd) -> None:
