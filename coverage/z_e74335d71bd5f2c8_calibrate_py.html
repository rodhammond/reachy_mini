1#!/usr/bin/env python3 

2"""Derive Reachy Mini speaker-EQ gains from hifiscan measurements. 

3 

4The plastic head shell colors the speaker output ("boxy": low-mid resonances + 

5comb peaks). A GStreamer ``equalizer-10bands`` on the daemon's speaker branch 

6corrects it; this offline tool computes the 10 per-band gains (dB) to put under 

7``speaker_eq_gains`` in the daemon config (``daemon_config.json``). It is not 

8used at runtime and users are unlikely to need it. See ``README.md``. 

9 

10Workflow: 

11 

121. ``uv pip install hifiscan`` and connect an *external* mic at a fixed 

13 listening position (not the robot's own mic; its XVF3800 DSP colors the 

14 measurement). Because we take a *differential*, the mic/amp/room coloration 

15 cancels, so a calibrated mic is not required, only that nothing moves. 

162. Run hifiscan and sweep the robot speaker twice, same volume/position, 

17 averaging several sweeps each (a single sweep is noisy): 

18 - shell **on** (assembled robot) -> hifiscan "save" -> ``corr_on.pkl`` 

19 - shell **off** (speaker in its bare case) -> ``corr_off.pkl`` 

203. Derive the gains:: 

21 

22 uv run python src/reachy_mini/tools/speaker_eq_calibration/calibrate.py \ 

23 gains corr_on.pkl corr_off.pkl 

24 

25 Put the printed list under ``speaker_eq_gains`` in the daemon config 

26 (``~/.config/reachy_mini/daemon_config.json``), or bake it into 

27 ``reachy_mini.media.audio_utils.DEFAULT_SPEAKER_EQ_GAINS``. 

284. Optionally render the calibration figure (needs matplotlib):: 

29 

30 uv run --with matplotlib python \ 

31 src/reachy_mini/tools/speaker_eq_calibration/calibrate.py plot \ 

32 corr_on.pkl corr_off.pkl corr_on_16k.pkl corr_off_16k.pkl \ 

33 -o docs/assets/speaker_eq_calibration.png 

34 

35Caveats: a 10-band graphic EQ has fixed octave bands, so a narrow resonance may 

36fall between centers; it corrects the magnitude response only (not time-domain 

37ringing or mechanical buzz); and on the 16 kHz voice path only bands <= 8 kHz 

38carry signal (top bands are zeroed via ``--zero-above``). 

39""" 

40 

41import argparse 

42 

43import numpy as np 

44import numpy.typing as npt 

45 

46# equalizer-10bands ISO octave band centers (Hz) and per-band gain range (dB). 

47BAND_CENTERS = ( 

48 29.0, 

49 59.0, 

50 119.0, 

51 237.0, 

52 474.0, 

53 947.0, 

54 1889.0, 

55 3770.0, 

56 7523.0, 

57 15011.0, 

58) 

59GAIN_MIN, GAIN_MAX = -24.0, 12.0 

60 

61_SMOOTH = 15.0 # hifiscan spectral smoothing before octave-band averaging 

62 

63 

64def bin_to_bands( 

65 freqs: npt.NDArray[np.float64], mag_db: npt.NDArray[np.float64] 

66) -> list[float]: 

67 """Average a magnitude-response curve (dB vs Hz) into the 10 EQ bands. 

68 

69 Each band takes the mean dB over a +/- half-octave window around its center. 

70 """ 

71 out = [] 

72 for center in BAND_CENTERS: 

73 lo, hi = center / 2**0.5, center * 2**0.5 

74 mask = (freqs >= lo) & (freqs < hi) 

75 out.append(float(np.mean(mag_db[mask])) if mask.any() else 0.0) 

76 return out 

77

79def band_gains_from_responses( 

80 freqs: npt.NDArray[np.float64], 

81 resp_on_db: npt.NDArray[np.float64], 

82 resp_off_db: npt.NDArray[np.float64], 

83 zero_above: float | None = None, 

84 max_boost: float = GAIN_MAX, 

85) -> list[float]: 

86 """Compute the 10 EQ band gains (dB) that cancel the shell's coloration. 

87 

88 ``resp_on_db`` / ``resp_off_db`` are the measured magnitude responses (dB) of 

89 the speaker with the head shell on and off. What the shell adds is:: 

90 

91 coloration(f) = resp_on(f) - resp_off(f) 

92 

93 and the correcting gain is its negative. We recenter by the median of the 

94 trusted bands (the two measurements differ by an arbitrary overall level we 

95 must not apply as broadband gain), clamp to the element's range, and zero 

96 bands above ``zero_above`` (untrusted treble and/or beyond the voice-path 

97 Nyquist). ``max_boost`` caps positive gains (noisy differentials can ask for 

98 unrealistic boosts that only add hiss and eat headroom). 

99 """ 

100 gains = bin_to_bands(freqs, resp_off_db - resp_on_db) # = -coloration 

101 

102 trusted = [ 

103 g 

104 for center, g in zip(BAND_CENTERS, gains) 

105 if zero_above is None or center <= zero_above 

106 ] 

107 if trusted: 

108 offset = float(np.median(trusted)) 

109 gains = [g - offset for g in gains] 

110 

111 ceiling = min(max_boost, GAIN_MAX) 

112 clamped = [min(max(g, GAIN_MIN), ceiling) for g in gains] 

113 if zero_above is not None: 

114 clamped = [ 

115 g if center <= zero_above else 0.0 

116 for center, g in zip(BAND_CENTERS, clamped) 

117 ] 

118 return clamped 

119

121def _load_response( 

122 path: str, smoothing: float = _SMOOTH 

123) -> tuple[npt.NDArray[np.float64], npt.NDArray[np.float64]]: 

124 """Return (freqs, magnitude_dB) from a hifiscan "save" (pickled Analyzer).""" 

125 import pickle 

126 

127 with open(path, "rb") as fh: 

128 analyzer = pickle.load(fh) 

129 spectrum = analyzer.spectrum(smoothing) 

130 return spectrum.x, spectrum.y 

131 

132 

133def _coloration( 

134 on: str, off: str 

135) -> tuple[npt.NDArray[np.float64], npt.NDArray[np.float64]]: 

136 """Return (freqs, coloration_dB) with the overall level removed. 

137 

138 Centered on the band-median (<=8 kHz) so it mirrors the net-zero EQ. 

139 """ 

140 freqs, resp_on = _load_response(on) 

141 freqs_off, resp_off = _load_response(off) 

142 col = resp_on - np.interp(freqs, freqs_off, resp_off) 

143 bands = bin_to_bands(freqs, col) 

144 med = float(np.median([g for c, g in zip(BAND_CENTERS, bands) if c <= 8000])) 

145 return freqs, col - med 

146 

147 

148def _cmd_gains( 

149 on: str, off: str, zero_above: float, max_boost: float, smoothing: float 

150) -> None: 

151 """Print the 10 EQ band gains and a daemon_config.json snippet.""" 

152 freqs_on, resp_on = _load_response(on, smoothing) 

153 freqs_off, resp_off = _load_response(off, smoothing) 

154 if not np.array_equal(freqs_on, freqs_off): # same chirp/rate -> same axis 

155 resp_off = np.interp(freqs_on, freqs_off, resp_off) 

156 

157 gains = band_gains_from_responses( 

158 freqs_on, resp_on, resp_off, zero_above=zero_above, max_boost=max_boost 

159 ) 

160 print("Band gains (dB):") 

161 for center, gain in zip(BAND_CENTERS, gains): 

162 print(f" {center:8.0f} Hz : {gain:+6.2f}") 

163 snippet = ", ".join(f"{g:.2f}" for g in gains) 

164 print(f'\ndaemon_config.json: "speaker_eq_gains": [{snippet}]') 

165

167def _cmd_plot(on44: str, off44: str, on16: str, off16: str, out: str) -> None: 

168 """Render the two-panel calibration figure (measured response + correction).""" 

169 import matplotlib 

170 

171 matplotlib.use("Agg") 

172 import matplotlib.pyplot as plt 

173 

174 ref, boxy, fix, col_c = "#5b6b7a", "#d1495b", "#2a9d8f", "#e9c46a" 

175 

176 # Panel 1: raw responses (44.1 kHz), level-matched over 100 Hz-8 kHz. 

177 freqs, resp_on = _load_response(on44) 

178 freqs_off, resp_off = _load_response(off44) 

179 resp_off = np.interp(freqs, freqs_off, resp_off) 

180 band = (freqs >= 100) & (freqs <= 8000) 

181 resp_on = resp_on - resp_on[band].mean() 

182 resp_off = resp_off - resp_off[band].mean() 

183 

184 # Panel 2: coloration (level-removed) + applied EQ (16 kHz voice set). 

185 fc44, col44 = _coloration(on44, off44) 

186 fc16, col16 = _coloration(on16, off16) 

187 f16, y16on = _load_response(on16) 

188 fo16, y16off = _load_response(off16) 

189 eq = band_gains_from_responses( 

190 f16, y16on, np.interp(f16, fo16, y16off), zero_above=8000.0 

191 ) 

192 edges, steps = [], [] 

193 for center, gain in zip(BAND_CENTERS, eq): 

194 edges += [center / 2**0.5, center * 2**0.5] 

195 steps += [gain, gain] 

196 

197 fig, (ax1, ax2) = plt.subplots(2, 1, figsize=(10, 8), sharex=True) 

198 

199 ax1.semilogx(freqs, resp_off, color=ref, lw=1.8, label="Shell off (bare case)") 

200 ax1.semilogx( 

201 freqs, resp_on, color=boxy, lw=1.8, label="Shell on (assembled — boxy)" 

202 ) 

203 ax1.set_title("Reachy Mini speaker: measured response (44.1 kHz, level-matched)") 

204 ax1.set_ylabel("Relative level (dB)") 

205 ax1.legend(loc="lower center", ncol=2, frameon=False) 

206 ax1.grid(True, which="both", alpha=0.25) 

207 

208 ax2.axhline(0, color="#999", lw=0.8) 

209 ax2.fill_between( 

210 fc16, col16, color=col_c, alpha=0.45, label="Shell coloration, 16 kHz" 

211 ) 

212 ax2.semilogx( 

213 fc44, col44, color="#8a6d3b", lw=1.0, ls=":", label="Shell coloration, 44.1 kHz" 

214 ) 

215 ax2.step(edges, steps, where="post", color=fix, lw=2.2, label="Applied 10-band EQ") 

216 for center in BAND_CENTERS: 

217 ax2.axvline(center, color="#ccc", lw=0.5, alpha=0.5) 

218 ax2.set_title("Shell coloration vs. applied correction (EQ ≈ −coloration)") 

219 ax2.set_xlabel("Frequency (Hz)") 

220 ax2.set_ylabel("Gain (dB)") 

221 ax2.set_xlim(20, 20000) 

222 ax2.legend(loc="lower center", ncol=2, frameon=False) 

223 ax2.grid(True, which="both", alpha=0.25) 

224 

225 fig.tight_layout() 

226 fig.savefig(out, dpi=130) 

227 print("saved", out)

230def main() -> None: 

231 """CLI: derive EQ gains from, or plot, two hifiscan measurements.""" 

232 parser = argparse.ArgumentParser( 

233 description="Reachy Mini speaker EQ calibration (offline reference tool)." 

234 ) 

235 sub = parser.add_subparsers(dest="cmd", required=True) 

236 

237 g = sub.add_parser("gains", help="print the 10 EQ band gains") 

238 g.add_argument("on", help="hifiscan measurement, shell ON") 

239 g.add_argument("off", help="hifiscan measurement, shell OFF") 

240 g.add_argument("--zero-above", type=float, default=8000.0) 

241 g.add_argument("--max-boost", type=float, default=12.0) 

242 g.add_argument("--smoothing", type=float, default=_SMOOTH) 

243 

244 p = sub.add_parser("plot", help="render the calibration figure (needs matplotlib)") 

245 p.add_argument("on44") 

246 p.add_argument("off44") 

247 p.add_argument("on16") 

248 p.add_argument("off16") 

249 p.add_argument("-o", "--out", default="docs/assets/speaker_eq_calibration.png") 

250 

251 args = parser.parse_args() 

252 if args.cmd == "gains": 

253 _cmd_gains(args.on, args.off, args.zero_above, args.max_boost, args.smoothing) 

254 else: 

255 _cmd_plot(args.on44, args.off44, args.on16, args.off16, args.out) 

256 

257 

258if __name__ == "__main__": 

259 main()
